var ComboKeysContext = require('../Combokeys-Context');
 
/**
 * Create a filter that will enable/disable callbacks from certain class names
 * @implements {ContextPlugin}
 * @param {string[]} tagNames A list of tag names
 * @constructor
 */
var TagCallbackFilter = function(classNames,stopCallbackOnMatch) {
	// clone the class names
	this._classNames = classNames.map(function(value) {
		return ' ' + value + ' ';
	});
 
	if (typeof stopCallbackOnMatch === 'undefined') {
		this._stop = true;
	} else {
		this._stop = stopCallbackOnMatch;
	}
 
};
 
TagCallbackFilter.prototype.stopCallback = function(evt, element) {
	var i;
	var className = ' ' + element.className + ' ';
	for (i = 0; i < this._classNames.length; i++) {
		if (className.indexOf(this._classNames[i]) > -1) {
			if (this._stop) {
				return ComboKeysContext.STOP_CALLBACK;
			} else { // we want to
				return ComboKeysContext.ALLOW_CALLBACK;
			}
		}
	}
};
 
module.exports = TagCallbackFilter;
